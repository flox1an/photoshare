Swipe pattern (from RESEARCH.md Pattern 7):
```typescript
import { useDrag } from '@use-gesture/react';
const bind = useDrag(({ swipe: [swipeX] }) => {
  if (swipeX === 1) onPrev();   // swipe right → previous
  if (swipeX === -1) onNext();  // swipe left → next
});
// Apply: <div {...bind()} style={{ touchAction: 'none' }}>
```

Keyboard pattern (from RESEARCH.md Pitfall 7):
```typescript
useEffect(() => {
  const handler = (e: KeyboardEvent) => {
    if (e.key === 'ArrowRight') onNext();
    if (e.key === 'ArrowLeft') onPrev();
    if (e.key === 'Escape') onClose();
  };
  window.addEventListener('keydown', handler);
  return () => window.removeEventListener('keydown', handler);
}, [onNext, onPrev, onClose]);  // stable callbacks required
```
